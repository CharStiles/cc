function setup() { createCanvas(600, 600); noStroke(); colorMode(HSB, 360, 100, 100, 100); } function draw() { background(245); let s = 120, t = frameCount * 0.5; for (let y = 0; y < height; y += s) { let hue = (y/s * 60 + frameCount * 0.8) % 360; for (let x = -s; x < width + s; x += s) { fill(hue, 70 + sin(frameCount*0.1)*10, 90, 85); push(); translate(x + s/2 + (t % s), y + s/2); rotate(sin(frameCount*0.02 + y/s)*0.1); triangle(-s/2, -s/2, s/2, -s/2, 0, s/2 - 10); pop(); } } }
